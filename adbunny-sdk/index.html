Publisher Embed Guide

Serve dist/adbunny.umd.cjs from your CDN, then publishers add:

<div id="ad-container" style="width:640px;height:360px"></div>
<script src="https://cdn.adbunny.in/adbunny.js"></script>
<script>
  var tag = new BunnyTag({
    container: '#ad-container',
    tags: [
      { url: 'https://ads.example.com/vast1.xml' },
      { url: 'https://ads.example.com/vast2.xml' },
    ],
    strategy: 'first-valid-media',
    timeout: 5000,
    autoplay: true,
    muted: false,
  });

  tag.on('statechange', function (e) {
    console.log('State:', e.state);
  });

  tag.on('adcomplete', function () {
    console.log('Ad completed');
  });

  tag.init();
</script>
